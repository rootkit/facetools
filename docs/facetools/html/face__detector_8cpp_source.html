67 std::vector<face> face_detector::detect(dlib::matrix<dlib::rgb_pixel>& image)
68 {
69  int max_size = std::max(image.nr(), image.nc());
70  int times_scaled = 0;
71 
72  while(max_size < params_.max_scaling_length && times_scaled < params_.max_scaling_times) {
73  dlib::pyramid_up(image);
74  max_size = std::max(image.nr(), image.nc());
75  ++times_scaled;
76  }
77 
78  if(params_.detector_type == face_detector_type_t::MMOD)
79  return mmod_detection_(image);
80  else
81  return frontal_face_detection_(image);
82 }

94 dlib::matrix<dlib::rgb_pixel> face_detector::downscale_image(const dlib::matrix<dlib::rgb_pixel>& input_image)
95 {
96  int max_dimension = std::max(input_image.nr(), input_image.nc());
97 
98  dlib::matrix<dlib::rgb_pixel> image;
99 
100  if(max_dimension > params_.max_scaling_length) {
101  float scaling = 1.0 * params_.max_scaling_length / (1.0 * max_dimension);
102  int resized_width = scaling * input_image.nc();
103  int resized_height = scaling * input_image.nr();
104  image = dlib::matrix<dlib::rgb_pixel>(resized_height, resized_width);
105  dlib::resize_image(input_image, image);
106  }
107  else {
108  image = input_image;
109  }
110 
111  return image;
112 }
