3  * then uses a metric based connectivity criterion to use for the Chinese

55 std::vector<embedding_t> face_recogniser::get_embedding(const std::vector<face>& input_faces)
56 {
57  unsigned int input_faces_size = input_faces.size();
58  std::vector<dlib::matrix<dlib::rgb_pixel>> faces(input_faces_size);
59 
60  std::vector<embedding_t> embeddings;
61 
62  if(params_.jitter_images) {
63  for(int i=0; i<input_faces_size; i++) {
64  auto jitter_stack = jitter_image_(input_faces[i].image);
65  dlib::matrix<float,0,1> face_descriptor = dlib::mean(dlib::mat(recogniser_(jitter_stack)));
66  embeddings.push_back(std::move(face_descriptor));
67  }
68  }
69  else {
70  for(int i=0; i<input_faces_size; i++)
71  faces[i] = input_faces[i].image;
72 
73  embeddings = recogniser_(faces);
74  }
75 
76  return embeddings;
77 }

126 std::vector<facelist_t> face_recogniser::get_chinese_whispers_clusters_(const std::vector<embedding_t>& embeddings)
127 {
128  size_t faces_size = embeddings.size();
129 
130  std::vector<dlib::sample_pair> edges;
131  for (size_t i = 0; i < faces_size; ++i)
132  for (size_t j = i+1; j < faces_size; ++j)
133  if (dlib::length(embeddings[i]-embeddings[j]) < params_.face_difference_threshold)
134  edges.push_back(dlib::sample_pair(i,j));
135 
136  std::vector<unsigned long> labels;
137  const auto num_clusters = dlib::chinese_whispers(edges, labels);
138  auto labels_size = labels.size();
139 
140  std::vector<bool> visited_face(faces_size);
141  auto unvisited_count = chinese_whispers_get_unvisited_faces_(visited_face, labels_size);
142 
143  std::vector<facelist_t> people(num_clusters + unvisited_count);
144 
145  for(size_t i=0; i<labels_size; ++i)
146  people[labels[i]].push_back(i);
147 
148  chinese_whispers_add_remaining_faces_(people, visited_face, faces_size, num_clusters);
149 
150  return people;
151 }

154 std::vector<dlib::matrix<dlib::rgb_pixel>> face_recogniser::jitter_image_(const dlib::matrix<dlib::rgb_pixel>& img)
155 {
156  thread_local dlib::random_cropper cropper;
157  cropper.set_chip_dims(150,150);
158  cropper.set_randomly_flip(true);
159  cropper.set_max_object_height(0.99999);
160  cropper.set_background_crops_fraction(0);
161  cropper.set_min_object_height(0.97);
162  cropper.set_translate_amount(0.02);
163  cropper.set_max_rotation_degrees(3);
164 
165  std::vector<dlib::mmod_rect> raw_boxes(1), ignored_crop_boxes;
166  raw_boxes[0] = dlib::shrink_rect(dlib::get_rect(img),3);
167  std::vector<dlib::matrix<dlib::rgb_pixel>> crops;
168 
169  dlib::matrix<dlib::rgb_pixel> temp;
170  for (int i = 0; i < 100; ++i) {
171  cropper(img, raw_boxes, temp, ignored_crop_boxes);
172  crops.push_back(std::move(temp));
173  }
174 
175  return crops;
176 }
