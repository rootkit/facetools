50 template <template <int,template<typename>class,int,typename> class block, int N, template<typename>class BN, typename SUBNET>
51 using residual_down = dlib::add_prev2<dlib::avg_pool<2,2,2,2,dlib::skip1<dlib::tag2<block<N,BN,2,dlib::tag1<SUBNET>>>>>>;

53 template <int N, template <typename> class BN, int stride, typename SUBNET>
54 using block = BN<dlib::con<N,3,3,1,1,dlib::relu<BN<dlib::con<N,3,3,stride,stride,SUBNET>>>>>;
