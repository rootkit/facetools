64 std::vector<std::string> facegrep::search(const std::vector<std::string> image_files)
65 {
66  std::vector<std::string> files_found;
67  for(auto& image_file : image_files)
68  {
69  auto faces = detector_->extract_faces(image_file);
70  auto embeddings = recogniser_->get_embedding(faces);
71  for(auto& candidate : embeddings)
72  if(face_matched_(template_embedding_, candidate))
73  files_found.push_back(image_file);
74  }
75 
76  return files_found;
77 }
